        // Counter Animation

        const counters = document.querySelectorAll('.counter');

        const speed = 200;

        const animateCounters = () => {

            counters.forEach(counter => {

                const updateCount = () => {

                    const target = +counter.getAttribute('data-target');

                    const count = +counter.innerText;

                    const inc = target / speed;

                    if (count < target) {

                        counter.innerText = Math.ceil(count + inc);

                        setTimeout(updateCount, 20);

                    } else {

                        counter.innerText = target + (target > 100 ? '+' : '');

                    }

                };

                updateCount();

            });

        }

        let counted = false;

        const statsSection = document.querySelector('.counter').closest('section');

        const observer = new IntersectionObserver((entries) => {

            if(entries[0].isIntersecting && !counted) {

                animateCounters();

                counted = true;

            }

        }, { threshold: 0.5 });

        observer.observe(statsSection);

        let currentLang = 'fr';

        

        // Typewriter Variables

        const typewriterElement = document.getElementById('typewriter');

        let phraseIndex = 0;

        let charIndex = 0;

        let isDeleting = false;

        let typeSpeed = 80;

        let typewriterTimeout; 


        function toggleLanguage() {

            currentLang = currentLang === 'fr' ? 'en' : 'fr';

            updatePageLanguage();

            resetTypewriter();

        }


        function updatePageLanguage() {

            // Update Toggle UI

            document.body.setAttribute('data-lang', currentLang);

            document.getElementById('btn-fr').classList.toggle('active', currentLang === 'fr');

            document.getElementById('btn-en').classList.toggle('active', currentLang === 'en');


            // Update Text Content

            const elements = document.querySelectorAll('[data-i18n]');

            elements.forEach(el => {

                const key = el.getAttribute('data-i18n');

                if (translations[currentLang][key]) {

                    el.textContent = translations[currentLang][key];

                }

            });

        }


        function resetTypewriter() {

            clearTimeout(typewriterTimeout);

            typewriterElement.classList.remove('typewriter-done');

            phraseIndex = 0;

            charIndex = 0;

            isDeleting = false;

            typewriterElement.textContent = '';

            type(); // Restart

        }


        // Typewriter Effect Logic

        function type() {

            const phrases = translations[currentLang].typewriter_words;

            

            if (phraseIndex === phrases.length - 1 && charIndex === phrases[phraseIndex].length) {

                typewriterElement.classList.add('typewriter-done'); 

                return; 

            }


            const currentPhrase = phrases[phraseIndex];

            

            if (isDeleting) {

                typewriterElement.textContent = currentPhrase.substring(0, charIndex - 1);

                charIndex--;

                typeSpeed = 30; 

            } else {

                typewriterElement.textContent = currentPhrase.substring(0, charIndex + 1);

                charIndex++;

                typeSpeed = 80; 

            }


            if (!isDeleting && charIndex === currentPhrase.length) {

                isDeleting = true;

                typeSpeed = 1500; 

            } else if (isDeleting && charIndex === 0) {

                isDeleting = false;

                phraseIndex++;

                typeSpeed = 300; 

            }


            typewriterTimeout = setTimeout(type, typeSpeed);

        }


        document.addEventListener('DOMContentLoaded', type);
